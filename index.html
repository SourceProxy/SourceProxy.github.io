const express = require('express'); const axios = require('axios'); const app = express(); const port = 3000; app.get('/api/request/:url', async (req, res) => { try { const response = await axios.get(req.params.url); res.send(response.data); } catch (error) { res.status(error.response ? error.response.status : 500).send(error.message); } }); app.listen(port, () => { console.log(`Server started on port ${port}`); });
